collections

  • List, Tuple, Dict에 대한 Python Built-in 확장 자료 구조(모듈)
  • 편의성, 실행 효율(메모리의 사용량을 줄이거나 속도를 좀 더 빠르게) 등을 사용자에게 제공합니다.
  • 아래의 여러 개의 모듈이 존재합니다.
from collections import deque
from collections import Counter
from collections import OrderedDict
from collections import defaultDict
from collections import namedtuple

OrderedDict

  • Dict와 달리, 데이터를 입력한 순서대로 dict를 반환함
  • 그러나 dict도 python 3.6 부터 입력한 순서를 보장하여 출력함
# 3.6 버전 이하의 파이썬
d = {}

dict['x'] = 100
dict['y'] = 200
dict['z'] = 300
dict['l'] = 500

for k, v in d.items():
	print(k, v)
# 출력
# l 500
# x 100
# y 200
# z 300
# 3.6 버전 이하의 파이썬
from collections import OrderedDict

ordered_dict = {}

ordered_dict['x'] = 100
ordered_dict['y'] = 200
ordered_dict['z'] = 300
ordered_dict['l'] = 500

for k, v in d.items():
	print(k, v)
# 출력
# x 100
# y 200
# z 300
# l 500

defaultdict

Dict type의 값에 기본 값을 지정, 신규값 생성시 사용하는 방법

from collections import defaultdict

d = defaultdict(object) # Default dictionary를 생성
d = defaultdict(lambda: 0) # Default 값을 0으로 설절함
print(d['first'])
# 출력
# 0
  • 하나의 지문에 각 단어들이 몇 개나 있는지 세고 싶을경우?
  • Text-mining 접근법 - Vector Space Model
text = " efj apwa dkwaowd woa jwdDDSdsdw Owjda ojawo ".lower().split()

def get_key(x):
	return x[1]

from collections import defaultdict

d = defaultdict(lambda : 0)
print(d)
# 출력
# defaultdict(<function __main__.<lambda>()>, {})

for word in text:
	d[word] += 1

print(d)
# 출력
# defaultdict(<function __main__.<lambda>()>, {'a': 0, 'press': 0, 'efj' : 1, ...})

from typing import Dict
def get_key(items: Dict[]) -> :
	return items[1]

sorted_dict = OrderedDict()
for i, v in sorted(d.items(), key=get_key, reverse=True): # value 값을 기준으로 내림차순 정렬
	sorted_dict[i] = v

namedtuple

Tuple 형태로 Data 구조체를 저장하는 방법으로, 저장되는 data의 variable을 사전에 지정해서 저장합니다.

from collections import namedtuple

Point = namedtuple("Point", ['x', 'y'])
print(Point)
# 출력
# __main__.Point

p = Point(11, y=22)
print(p[0], p[1])
# 출력
# (11, 22)

x, y = p # unpacking
print(x, y)
# 출력
# 11, 22

print(p.x + p.y)
# 출력
# 33

print(Point(x=11, y=22))
# 출력
# Point(x=11, y=22)
