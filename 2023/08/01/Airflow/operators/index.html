BashOperator

from airflow.operators.bash_operator import BashOperator

# case 1: 일반적인 방법으로 bash command 실행
bash_task1 = BashOperator(
    task_id: str = "bash_task1",
    bash_command: str =
    # 실행할 command 또는 set of commands 또는 bash script(‘.sh’여야 함)
    """
        hdfs dfs -mkdir -p /forex && \
        hdfs dfs -put -f $AIRFLOW_HOME/dags/files/forex_rates.json /forex
    """
)

""""""
# case 2: template으로 bash command 실행
templated_command = """
    {% for i in range(5) %}
        echo "{{ ds }}"
        echo "{{ macros.ds_add(ds, 7)}}"
        echo "{{ params.my_param }}"
    {% endfor %}
"""

bash_task2 = BashOperator(
    task_id: str ="bash_task2",
    bash_command: str = templated_command
)

PythonOperator

from airflow.operators.python_operator import PythonOperator

def my_sleeping_function(random_base):
    """This is a function that will run within the DAG execution"""
    time.sleep(random_base)

    i = 5

    sleeping_task = PythonOperator(
        task_id: str = f"sleep_for_{i}",
        python_callable: Callable = my_sleeping_function, # 호출 가능한 객체에 대한 참조
        # op_args: Collection[Any] | None  # 함수 호출 시 펼쳐질 positional argument의 list
        op_kwargs: Mapping[str, Any] | None = {"random_base": i / 10} # 함수 호출시 펼쳐질 keyword argument의 dictionary
    )

BranchPythonOperator

from airflow.operators.python_operator import BranchPythonOperator

# 다음 작업은 파이썬 함수 check_api의 결과 값에 따라 달라집니다.
check_api = BranchPythonOperator(
    task_id: str ='check_api',
    python_callable: Callable = check_api
)

check_api >> none >> save

# API에 따라 동적으로 task를 생성합니다.
for api in IP_GEOLOCATION_APIS:

    )

    check_api >> process >> save

BranchDateTimeOperator

from airflow.operators.datetime import BranchDateTimeOperator

is_in_timeframe = BranchDateTimeOperator(
    task_id="is_in_timeframe",
    follow_tasks_ids_if_true=["move_forward"],
    follow_tasks_ids_if_false=["end"],
    target_lower=time(10, 0, 0),
    target_upper=time(10, 0, 0),
    use_task_execution_date=True
)

HttpSensor

from airflow.sensors.http_sensor import HttpSensor

# Checking if forex rates are avaiable
# TODO: Check SSL
my_http_sensor = HttpSensor(
    task_id="my_http_sensor", 
    method="GET", # 사용할 HTTP 요청 방법
    http_conn_id="forex_api", # 센서를 실행하기 위한 HTTP 연결
    endpoint="latest", # 전체 URL의 상대적인 부분
    response_check=lambda response: "rates" in response.text, #  '요청' 응답 객체에 대한 검사하는 함수
    poke_interval=5, # 센서가 조건을 확인하는 주기
    timeout=20 # 센서가 조건을 확인 할 최대 시간
)

SimpleHttpOperator

from airflow.operators.http_operator import SimpleHttpOperator

# get forex rates of JPY and push them into XCOM
get_forex_rate_EUR = SimpleHttpOperator(
    task_id='get_forex_rate_EUR',
    method='GET', # 사용할 HTTP 메소드, 기본값 = 'POST'
    priority_weight=1,
    pool='forex_api_pool',
    http_conn_id='forex_api',
    endpoint='/latest?base=EUR',
    xcom_push=True
)

FileSensor

from airflow.contrib.sensors.file_sensor import FileSensor

# 관찰할 파일이 있는 지 확인
# Web에서 fs_conn_id를 필수적으로 연결해서 경로를 지정해야 한다
# in the extra parameter as JSON
is_file_available = FileSensor(
    task_id="is_file_available",
    fs_conn_id="forex_path", # 파일 (경로) 연결 ID에 대한 참조
    filepath="forex_currencies.csv", # 파일 또는 폴더 이름 (fs_conn_id에 설정 기본 경로를 기준으로 상대경로)
    poke_interval=5,
    timeout=20
)

HiveOperator

from airflow.operators.hive_operator import HiveOperator

# HiveOperator로 테이블 만들기
creating_hive_table = HiveOperator(
    task_id="creating_hive_table",
    hive_cli_conn_id="hive_conn", # Hive 데이터베이스에 대한 참조
    hql="""
        CREATE EXTERNAL TABLE IF NOT EXISTS forex_rates(
            base STRING,
            last_update DATE,
            eur DOUBLE,
            usd DOUBLE,
            nzd DOUBLE,
            gbp DOUBLE,
            jpy DOUBLE,
            cad DOUBLE
        )
        ROW FORMAT DELIMITED
        FIELDS TERMINATED BY ','
        STORED AS TEXTFILE
    """
)

SparkSubmitOperator

from airflow.contrib.operators.spark_submit_operator import SparkSubmitOperator

# 데이터 처리를 위해 Spark Job 실행하기
spark_processing = SparkSubmitOperator(
    task_id="spark_processing",
    application="/usr/local/airflow/dags/scripts/spark_processing.py", # 작업으로 제출된 응용 프로그램, jar 또는 py 파일
    conn_id="spark_conn", # Airflow 관리에서 구성된 연결 ID. 잘못된 연결 ID가 제공되면 기본적으로 yarn으로 설정
    verbose=False # 디버깅을 위해 verbose 플래그를 spark-submit 프로세스에 전달할지 여부
)

PostgresOperator

from airflow.operators.postgres_operator import PostgresOperator

create_table = PostgresOperator(
    task_id="create_table",
    sql: str = "CREATE TABLE test_table (table_value TEXT NOT NULL, PRIMARY KEY (table_value));", # SQL code 
    postgres_conn_id: str = "postgres" # 특정 PostgreSQL 데이터베이스를 참조하는 Postgres 연결 ID.
)

# PostgresOperator를 상속받아 sql에 param을 넣을 수 있게 재구현
class CustomPostgresOperator(PostgresOperator):
    template_fields = ("sql", "parameters")

create_table = CustomPostgresOperator(
    task_id="create_table",
    postgres_conn_id="postgres"
    sql=[
        "sql/CREATE_TABLE_MY_TABLE.sql",
        "SELECT * FROM my_table"
    ],
    parameters={
        "filename": '{{ ti.xcom_pull(task_ids=["my_task"]) }}'
    }
)
