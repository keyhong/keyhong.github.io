HiveQL 정리

Hive

DML

테이블 목록 보기

SHOW TABLES;

파티션 정보 보기

SHOW PARTITIONS <테이블>;

테이블 통계 정보 보기

ANALYZE TABLE <테이블> PARTITION (<파티션 컬럼> = <조건>) COMPUTE STATISTICS;

파티션별 테이블 총 개수 보기

SELECT <파티션 컬럼>, COUNT(*) FROM <테이블> GROUPBY <파티션 컬럼> ORDER BY 1;

문자열 함수

☆ CONCAT (string|binary A, string|binary B…)

두 문자열을 결합하여 반환한다.

SELECT CONCAT('Hello', '-', 'World');
-- Return
-- 'Hello-World'

LENGTH (string A)

컬럼에 있는 개별 값들의 문자열의 길이를 반환한다.

SELECT LENGTH('England');
-- Return
-- 7

LOWER (string A)

모든 대문자를 소문자로 변환한 문자열을 반환한다.

SELECT LOWER('England');
-- Return
-- england

UPPER (string A)

모든 소문자를 대문자로 변환한 문자열을 반환한다.

SELECT UPPER('England');
-- Return
-- ENGLAND

LPAD (string str, int len, string pad)

원하는 문자를 문자열의 길이를 만족할 때까지 왼쪽부터 채워서 반환한다.

SELECT LPAD('England', 10, 'v');
-- Return
-- vvvEngland

SELECT LPAD('Wales', 10, 'v');
-- Return
-- vvvvvWales

RPAD (string str, int len, string pad)

원하는 문자를 문자열의 길이를 만족할 때 까지 오른쪽부터 채워서 반환한다.

SELECT RPAD('England', 10, 'v');
-- Return
-- Englandvvv

SELECT RPAD('Wales', 10, 'v');
-- Return
-- Walesvvvvv

LTRIM (string A)

문자열 안의 공백을 왼쪽부터 모두 잘라내어 반환한다.

SELECT LTRIM('       keyhong');
-- Return
-- 'keyhong'

RTRIM (string A)

문자열 안의 공백을 오른쪽부터 모두 잘라내어 반환한다.

SELECT 	RTRIM('keyhong       ');
-- Return
-- 'jivesh'

REPEAT (string str, int n)

원하는 문자열을 원하는 개수만큼 반복해서 결합한 문자열을 반환한다.

SELECT REPEAT('England', 2);
-- Return
-- EnglandEngland

REVERSE (string A)

문자열을 뒤집어서 반환한다.

SELECT REVERSE('England');
-- Return
-- dnalgnE

☆ SPLIT (string str, string pat)

구분 기호나 특수 문자를 기준으로 문자열을 분할한다.

SELECT SPLIT('hive:hadoop', ':');
-- Return
-- ["hive", "hadoop"]

☆ SUBSTR (string|binary A, int start)

문자열에서 주어진 인덱스만큼의 문자열을 빼내어 반환한다. 하이브에서 첫번째 문자의 인덱스는 1부터 시작한다.

SELECT SUBSTR('hive is querying tool', 4);
-- Return
-- 'e is querying tool'

SELECT SUBSTR('hive is querying tool', 4, 3);
-- Return
-- 'e i'

☆ INSTR (string str, string substr)

문자열에서 주어진 특정 문자나 문자열의 위치를 인덱스로 반환한다.

SELECT INSTR('hive is querying tool', 'e');
-- Return
-- 4

☆ RLIKE

Hive에서 제공하는 특수기능으로 A의 하위 문자열이 B와 일치하면 true로 평가하여 필요 데이터만 추출할 수 있다. 정규표현식과 함께 사용하면 유용하다. 탐색할 문자열의 앞이나 뒤에 '%', '*' 등의 특수문자를 사용할 수 있다. Right Like의 약어이다.

- 일반적인 단어 매칭 확인한다.
SELECT 'hadoop' RLIKE 'ha';
/* Return
_c0
true
*/

-- 정규표현식을 이용해서 매칭 확인한다.
SELECT 'hadoop' RLIKE 'ha*';
/* Return
_c0
true
*/

-- 공백이 있을 때 Trim 없이도 유용하게 사용할 수 있다. 
SELECT 'hadoop       ' RLIKE 'hadoop';
/* Return
_c0
true
*/

-- 탐색할 문자열이나 탐색 기준 문자열이 null일 경우, NULL을 반한다.
SELECT null RLIKE 'hadoop';
/* Return
_c0
NULL
*/

SELECT 'hadoop' RLIKE null
/* Return
_c0
NULL
*/

날짜 함수

UNIX_TIMESTAMP (string date)

타임스탬프 인수를 UNIX 시간대(‘1970-01-01 00:00:00’)에서 몇 초나 지났는 지 epoch를 정수로 반환한다.

SELECT UNIX_TIMESTAMP('2023-12-13 00:00:00');
-- 초단위로 결과 반환

FROM_UNIXTIME (bigint unixtime[, string format])

초 단위의 인수를 UNIX 기준 시간대( ‘1970-01-01 00:00:00’ )로 부터 지난 타임스탬프를 반환한다.

SELECT FROM_UNIXTIME(1234145);
-- Return
-- 1970-01-15 12:19:05

☆ TO_DATE (string timestamp)

timestamp의 date 부분을 ‘yyyy-MM-dd’ 형식의 문자열로 반환한다.

SELECT TO_DATE('2000-01-01 10:20:30');
-- Return
-- '2000-01-01'

YEAR (string date)

date의 연도를 정수로 반환한다.

SELECT YEAR('2000-01-01 10:20:30');
-- Return
-- 2000

MONTH (string date)

date의 월을 정수로 반환한다.

SELECT MONTH('2000-01-01 10:20:30');
-- Return
-- 1

DAY (string date), DAYOFMONTH (date)

date의 일을 정수로 반환한다.

SELECT DAY('2000-03-01 10:20:30');
-- Return
-- 1

HOUR (string date)

date의 시간을 정수로 반환한다.

SELECT HOUR('2000-03-01 10:20:30');
-- Return
-- 10

MINUTE (string date)

timestamp의 분을 정수로 반환한다.

SELECT MINUTE('2000-03-01 10:20:30');
-- Return
-- 20

SECOND (string date)

timestamp의 초를 정수로 반환한다.

SELECT SECOND('2000-03-01 10:20:30');
-- Return
-- 30

WEEKOFYEAR (string date)

date가 해당 연도에서 몇 번째 주인지 정수로 반환한다.

SELECT WEEKOFYEAR('2000-03-01 10:20:30');
-- Return
-- 9

DATEDIFF(string enddate, string startdate)

주어진 두 date의 차이 일수를 정수로 반환한다.

SELECT DATEDIFF('2000-03-01', '2000-01-10');
-- Return
-- 51

DATE_ADD (string startdate, int days)

지정한 날짜에서 인자로 넣은 일의 수 만큼을 더한 날짜를 문자열로 반환한다.

SELECT DATE_ADD('2000-03-01', 5);
-- Return
-- '2000-03-06'

DATE_SUB (string startdate, int days)

지정한 날짜에서  인자로 넣은 일의 수 만큼을 뺀 날짜를 문자열로 반환한다.

SELECT DATE_SUB('2000-03-01', 5);
-- Return
-- '2000-02-25'

숫자 연산 함수

CEIL (double a)

소수점을 올림하여 정수로 반환한다.

SELECT CEIL(9.5);

-- Return
-- 10

FLOOR (double a)

소수점을 내림하여 정수로 반환한다.

SELECT FLOOR(10.9);

-- Return
-- 10

☆ ROUND (double a[, int d])

소수를 원하는 자리에서 반올림하여 반환한다.

SELECT ROUND(123.456, 2);
-- Return
-- 123.46

RAND ([,int seed])

랜덤한 난수를 반환한다.

SELECT RAND();
-- Return
-- 랜덤한 난수

SELECT RAND(23);
-- Return
-- 랜덤한 난수 (단, seed를 넣었기 때문에 여러 번 실행해도 동일한 난수를 반환한다.)

조건문 함수

☆ IF (boolean testCondition, T valueTrue, T valueFalseOrNull)

조건식을 만족하면 valueTrue를, 만족하지 않으면 valueFalseOrNull을 수행하여 반환한다.

SELECT IF(col3='England', col1, col4) FROM table;1
-- Return 
-- 499
-- 501
-- 500
-- 139000
-- 503
-- 504

☆ CASE a WHEN b THEN c [WHEN d THEN e]* [ELSE f] END

CASE fruit
     WHEN 'apple' THEN 'The owner is apple'
     WHEN 'orange' THEN 'The owner is orange'
     ELSE 'It is another Fruit'
END

☆ CASE WHEN a THEN b [WHEN c THEN d]* [ELSE e] END

CASE WHEN fruit = 'apple' THEN 'The owner is apple'
     WHEN fruit ='orange' THEN 'The owner is orange'
     ELSE 'It is another Fruit'
END

ISNULL (a)

컬럼 값이 NULL이면 true를, NULL이 아니면 false를 반환한다.

SELECT ISNULL(col1) FROM <table>;
-- Return
-- false
-- false
-- false

SNOTNULL (a)

ISNULL()과는 반대로 컬럼 값이 NULL이면 false를, NULL이 아니면 true를 반환한다.

SELECT ISNOTNULL(col1) FROM <table>;
-- Return
-- true
-- true
-- true

COALESCE (T v1, T v2, …)

주어진 컬럼들을 순서대로 탐색하여 NULL이 아닌 값이 있다면 반환한다.

SELECT COALESCE(datefield1, datefield2, datefield3) AS first_date_found FROM tblDates;
-- Return
-- 1993-06-04
-- 2005-06-12

☆ NVL (T value, T default_value)

컬럼 값이 NULL이면 default_value를 반환하고, NULL이 아니면 본래 값을 반환한다.

SELECT supplier_id, NVL(supplier_desc, 	supplier_name)
FROM suppliers;
-- supplier_desc 컬럼 값이 NULL이면 supplier_name의 컬럼 값을
-- 그렇지 않으면 supplier_desc의 컬럼 값을 반환

순위 함수

RANK()

동점자 이후 동점자 수만큼 순위를 건너 뛰고 다시 순위를 부여한다.

  • 정렬된 파티션 내에서 순위를 제공한다.
  • 동점자에게는 같은 순위를 제공하고, 다음 순위는 건너 뛴다.
  • RANK() 함수는 전체 데이터 세트에 대해 수행이 된다.

-- 기본 테이블
SELECT * FROM <table>;
/*
-----------------
| col1   | col2 |
-----------------
| John   | 200  |
| Albert | 180  |
| Mark   | 100  |
| Frank  | 150  |
| Mark   | 130  |
| John   | 200  |
| Lui    | 200  |
| John   | 180  |
|        | NULL |
-----------------
*/

-- RANK()
SELECT col1, col2, RANK() OVER(ORDER BY col2 DESC) AS rank
FROM <table>;
/* Return
------------------------
| col1   | col2 | rank |
------------------------
| John   |  200  | 1   |    
| John   |  200  | 1   |
| Lui    |  200  | 1   |
| John   |  180  | 4   |
| Albert |  180  | 4   |
| Frank  |  150  | 6   |
| Mark   |  130  | 7   |
| Mark   |  100  | 8   |
|        |  NULL | 9   |
------------------------
*/

DENSE_RANK()

동점자 이후 동점자 수만큼 순위를 건너 뛰고 다시 순위를 부여한다.

RANK()와 유사하다. 차이점은 동점자에게 같은 순위를 제공하고, 다음 순위가 연속적이다.

-- 기본 테이블
SELECT * FROM <table>;
/*
-----------------
| col1   | col2 |
-----------------
| John   | 200  |
| Albert | 180  |
| Mark   | 100  |
| Frank  | 150  |
| Mark   | 130  |
| John   | 200  |
| Lui    | 200  |
| John   | 180  |
|        | NULL |
-----------------
*/

-- DENSE_RANK()
SELECT col1, col2, DENSE_RANK() OVER(ORDER BY col2 DESC) as rank
FROM <table>;
/* Return
------------------------
| col1   | col2 | rank |
------------------------
| John   | 200  |    1 |    
| John   | 200  |    1 |
| Lui    | 200  |    1 |
| John   | 180  |    2 |
| Albert | 180  |    2 |
| Frank  | 150  |    3 |
| Mark   | 130  |    4 |
| Mark   | 100  |    5 |
|        | NULL |    6 |
------------------------
*/

ROW_NUMBER()

동점자에 상관없이 순서대로 순위를 부여한다.

  • 파티션 내의 각 행에 고유번호를 할당한다.

  • 같은 값에도 동일한 순위를 부여하지 않고, 순위를 건너 뛰지도 않는다.

-- 기본 테이블
SELECT * FROM <table>;
/*
-----------------
| col1   | col2 |
-----------------
| John   | 200  |
| Albert | 180  |
| Mark   | 100  |
| Frank  | 150  |
| Mark   | 130  |
| John   | 200  |
| Lui    | 200  |
| John   | 180  |
|        | NULL |
-----------------
*/

-- ROW_NUMBER()
SELECT col1, col2, ROW_NUMBER() OVER(ORDER BY col2 DESC) AS rownum
FROM <table>;

/* Return
--------------------------
| col1   | col2 | rownum |
--------------------------
| John   | 200  |      1 |
| John   | 200  |      2 |
| Lui    | 200  |      3 |
| John   | 180  |      4 |
| Albert | 180  |      5 |
| Frank  | 150  |      6 |
| Mark   | 130  |      7 |
| Mark   | 100  |      8 |
|        | NULL |      9 |
--------------------------
*/

WINDOW + ROW_NUMBER()

동점자에 상관없이 파티션절의 의해 만들어진 파티션별 순서에 차례대로 순위를 부여한다. Window Function PARTITION BY ~ ORDER BY ~와 같이 사용하여 굉장히 강력한 기능을 제공한다. PARTITION BY는 RANK()나 DENSE_RANK()에도 사용 가능하다.

-- 기본 테이블
SELECT * FROM <table>;
/*
-----------------
| col1   | col2 |
|---------------|
| John   | 200  |
| Albert | 180  |
| Mark   | 100  |
| Frank  | 150  |
| Mark   | 130  |
| John   | 200  |
| Lui    | 200  |
| John   | 180  |
|        | NULL |
-----------------
*/

-- WINDOW + ROW_NUMBER()
SELECT col1, col2, ROW_NUMBER() OVER(PARTITION BY col1 ORDER BY col2 DESC) AS rownum
FROM <table>;
/* Return
--------------------------
| col1   | col2 | rownum |
--------------------------
|        | NULL |      1 |
| Albert | 180  |      1 |
| Frank  | 150  |      1 |
| John   | 200  |      1 |
| John   | 200  |      2 |
| John   | 180  |      3 |
| Lui    | 200  |      1 |
| Mark   | 130  |      1 |
| Mark   | 100  |      2 |
--------------------------
*/

피벗함수

Explode()

컬럼의 값에 있는 배열의 모든 요소를 별도의 행으로 생성한다. 데이터가 많으면 행이 무수히 많아져 크기가 급속도로 커지므로 사용시에 항상 Out Of Memory 에러에 주의가 필요하다. 기본 테이블의 col2 컬럼에 EXPLODE()를 적용한 결과는 아래와 같다. 배열 안의 요소 값이 개별 행으로 분리되어 나온다.

-- 기본 테이블
SELECT * FROM <table>;
/*
--------------------
| col1 | col2      |
-------------------|
| A    | [1, 2, 5] |
| B    | [9, 2, 6] |
| C    | [7]       |
| D    | [6, 9]    |
| E    | [0, 3]    |
--------------------
*/

-- EXPLODE()
SELECT EXPLODE(col2) AS mycol FROM <table>;
/*
---------
| mycol |
---------
| 1     |
| 2     |
| 5     |
| 9     |
| 2     |
| 6     |
| 7     |
| 6     |
| 9     |
| 0     |
| 3     |
---------
*/

Lateral View

Explode()는 분해한 대상 컬럼만 조회 할 수 있는 단점이 있다. 분해된 컬럼 이외에 다른 컬럼을 함께 조회하려면 Lateral View를 같이 사용해야 한다. 분해된 컬럼의 결과는 가상 테이블(Virtual Table)에 임시 저장되며, 기존 테이블의 가상 테이블의 결합을 통해 join된 결과 테이블을 얻는다. col2을 explode()로 분해하고 LATERAL VIEW를 사용하여 dummy_col2 라는 가상 테이블에 저장한다. 이 가상 테이블을 기본 테이블 <table>과 결합하여 두 컬럼이 join된 결과 테이블을 도출한다.

-- 기본 테이블
SELECT * FROM <table>;

/* 
----------------------------------------------
| col1            | col2                     |
|--------------------------------------------|
| Salmon Rushdie  | [Grimus, Shame, Fury]    |
| Thomas Otway    | [Don Carlos, The Orphan] |
| Ben Jonson      | [Volpone, Epicene]       |
| John Milton     | [Arcades, Comus]         |
----------------------------------------------
*/

-- EXPLODE() + LATERAL VIEW
SELECT col1, col2
FROM <table>
LATERAL VIEW EXPLODE(col2) dummy AS dummy_col2;

/*
-------------------------------
| col1           | col2       |
|-----------------------------|
| Salmon Rushdie | Grimus     |
| Salmon Rushdie | Shame      |
| Salmon Rushdie | Fury       |
| Thomas Otway   | Don Carlos |
| Thomas Otway   | The Orphan |
| Ben Jonson     | Volpone    |
| Ben Jonson     | Epicene    |
| John Milton    | Arcades    |
| John Milton    | Comus      |
-------------------------------
*/

LATERAL VIEW for Map

Map(key:value) 타입에도 EXPLODE()와 LATERAL VIEW를 이용하면 key와 value를 각각의 컬럼으로 분리할 수 있다. 사용 방법은 LATERAL VIEW의 alias를 key,value로 준다. 그러면 key와 value를 분리한 별도의 테이블을 얻을 수 있다.

-- 기본 테이블
SELECT * FROM <table>;
/* 
-----------------
| col1          |
-----------------
| {"12": "abc"} |
| {"34": "def"} |
| {"56": "ghi"} |
| {"78": "jkl"} |
-----------------
*/

-- EXPLODE() + LATERAL VIEW
SELECT key, value
FROM <table>
LATERAL VIEW EXPLODE(col1) dummy AS key,value;

/* Return

12    abc
34    def
56    ghi
78    jkl
*/
